DQL: Lekérdezés

SELECT megjeleno_oszlopok_neve
FROM tabla(k)
WHERE szurofeltetel(ek)
GROUP BY mezonev(ek)
HAVING szurofeltetel(ek)
ORDER BY mezonev(ek) ASC | DESC
LIMIT melyik_elemtol, darabszam;
